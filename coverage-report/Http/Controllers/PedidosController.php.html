268            $pedidoPendiente = Pedidos::where('cliente_id', $user->id)
269                ->where('status', Pedidos::STATUS_PENDIENTE)
270                ->with('detallespedidos') // Carga los detalles para verificar si está vacío y calcular total.
271                // ->lockForUpdate() // Opcional: Bloquea la fila del pedido para evitar modificaciones concurrentes durante la transacción. Útil en sistemas de alta concurrencia.

276            if ($pedidoPendiente->detallespedidos->isEmpty()) {
